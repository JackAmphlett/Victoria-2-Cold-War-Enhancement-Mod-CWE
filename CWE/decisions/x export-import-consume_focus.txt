changefocus = {
	picture = "changefocus"
	potential = {
		war = no
		OR = { has_country_flag = importfocus has_country_flag = exportfocus has_country_flag = consumefocus }
	}
	alert = no
	allow = {
		OR = {
			has_country_flag = importfocus
			has_country_flag = exportfocus
			has_country_flag = consumefocus
		}
		OR = {
			ai = no
			AND = { is_secondary_power = yes NOT = { has_country_flag = exportfocus } }
			AND = { is_greater_power = yes NOT = { has_country_flag = exportfocus } }
		}
	}
	effect = {
		clr_country_flag = importfocus
		clr_country_flag = exportfocus
		clr_country_flag = consumefocus
		remove_country_modifier = spchigh
		remove_country_modifier = spcmedium
		remove_country_modifier = spclow
		clr_country_flag = spchigh
		clr_country_flag = spcmedium
		clr_country_flag = spclow
		any_owned = { remove_province_modifier = importsubstitutionfocus }
		any_owned = { remove_province_modifier = exportfocus }
		any_owned = { remove_province_modifier = consumefocus }
	}
	ai_will_do = {
		factor = 1
	}
}

importfocus = {
	picture = "importfocus"
	potential = {
		war = no
		NOT = { has_country_flag = importfocus }
		NOT = { has_country_flag = exportfocus }
	}
	allow = {
		OR = {
		ai = no
		rank = 30
	}
}
	effect = {
		set_country_flag = importfocus
		any_owned = { limit = { is_colonial = no } add_province_modifier = { name = importsubstitutionfocus duration = -1 } }
	}
	ai_will_do = {
		factor = 1
	}
}

	picture = "consumefocus"

		war = no
		NOT = { has_country_flag = exportfocus }

	}

		OR = {
			neoliberalism = 1
